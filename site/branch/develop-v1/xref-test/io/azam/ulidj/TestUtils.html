66    public static byte[] incrementBytes(byte[] bytes) {
67      if (bytes == null || bytes.length != ULID.ENTROPY_LENGTH)
68        return null;
69      byte[] value = new byte[ULID.ENTROPY_LENGTH];
70  
71      boolean carry = true;
72      for (int i = ULID.ENTROPY_LENGTH - 1; i >= 0; i--) {
73        byte work = bytes[i];
74        if (carry) {
75          work = (byte) (work + 0x01);
76          carry = bytes[i] == (byte) 0xff && carry;
77        }
78        value[i] = work;
79      }
80      // Last byte has carry over
81      if (carry) {
82        return null;
83      }
84  
85      return value;
86    }

88    public static String bytesToInitializer(byte[] bytes) {
89      if (bytes == null)
90        return "null";
91      StringBuilder sb = new StringBuilder();
92      sb.append("new byte[] {");
93      for (int i = 0; i < bytes.length; i++) {
94        sb.append(String.format("(byte) 0x%02x", bytes[i]));
95        if (i < bytes.length - 1)
96          sb.append(",");
97      }
98      sb.append(" }");
99      return sb.toString();
100   }

112   public static void assertEntropyIsIncremented(String prev, String next) {
113     assertEquals(prev.length(), next.length());
114     boolean charMustBeSame = false;
115     for (int i = next.length() - 1; i >= 0; i--) {
116       if (!charMustBeSame) {
117         // We assume only ASCII characters, so charAt is good enough here.
118         int nextCharIndex = indexOfElement(C, next.charAt(i));
119         int prevCharIndex = indexOfElement(C, prev.charAt(i));
120         assertTrue("Next ULID value contains invalid character", nextCharIndex >= 0);
121         assertEquals("Next base64 char is wrong", ((nextCharIndex + C.length - 1) % C.length),
122             prevCharIndex);
123         charMustBeSame = nextCharIndex < C.length && nextCharIndex > 0;
124       } else {
125         assertEquals(next.charAt(i), prev.charAt(i));
126       }
127     }
128   }

130   public static void assertEntropyIsIncremented(byte[] prev, byte[] next) {
131     assertEquals(prev.length, next.length);
132     assertArrayEquals(TestUtils.incrementBytes(ULID.getEntropyBinary(prev)),
133         ULID.getEntropyBinary(next));
134   }

136   public static int indexOfElement(char[] arr, char e) {
137     if (arr != null && arr.length > 0) {
138       for (int i = 0; i < arr.length; i++) {
139         if (arr[i] == e)
140           return i;
141       }
142     }
143     return -1;
144   }
