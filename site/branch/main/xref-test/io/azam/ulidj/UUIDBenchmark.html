21  package io.azam.ulidj;
22  
23  import java.util.UUID;
24  
25  import org.openjdk.jmh.annotations.Benchmark;
26  import org.openjdk.jmh.infra.Blackhole;
27  
28  public class UUIDBenchmark {
29    public static final String TEST_VALUE = "00526084-9560-7fff-ffff-ffffffffffff";
30  
31    @Benchmark
32    public void randomUUID(Blackhole blackhole) {
33      blackhole.consume(UUID.randomUUID());
34    }
35  
36    @Benchmark
37    public void fromString(Blackhole blackhole) {
38      blackhole.consume(UUID.fromString(TEST_VALUE));
39    }
40  }
