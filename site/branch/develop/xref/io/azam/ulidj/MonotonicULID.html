21  package io.azam.ulidj;
22  
23  import java.security.SecureRandom;
24  import java.util.Random;
25  
26  /**
27   * Monotonic instance of ULID. ULID spec defines monotonicity behavior as if a ULID is to be
28   * generated in the same millisecond, the entropy(random) component is to be incremented by 1-bit in
29   * the least significant bit with carryover.<br>
30   * <br>
31   * In order to achieve monotonicity, the state for timestamp and its last generated entropy must be
32   * persisted so that the next ULID generated within the same millisecond can generate an incremented
33   * byte array for its entropy.<br>
34   * <br>
35   * This class achieves that via behaving as a factory class that generates ULID strings, binary and
36   * ULID instances with the last timestamp and entropy persisted in the instance, and generates a new
37   * ULID based on that. It guarantees the monotonicity of ULID (string/binary/ULID instance) if
38   * called with the same {@link io.azam.ulidj.MonotonicULID} instance within the JVM.<br>
39   * <br>
40   * Normally you would want to create a singleton bean of this class in your application and use that
41   * to generate ULIDs.<br>
42   * <br>
43   * Monotonic ULID examples:<br>
44   *
45   * <pre>
46   * // Using default MonotonicULID static instance backed by SecureRandom
47   * String ulid1 = MonotonicULID.random();
48   * // Subsequent calls generates monotonic ULID
49   * String ulid2 = MonotonicULID.random();
50   * byte[] ulid3 = MonotonicULID.randomBinary();
51   * ULID ulid4 = MonotonicULID.randomULID();
52   *
53   * // Create a new independent instance of MonotonicULID using default Random
54   * // instance backed by SecureRandom
55   * MonotonicULID monotonicUlid = new MonotonicULID();
56   * String ulid5 = monotonicUlid.generate();
57   * // Subsequent calls generates monotonic ULID
58   * String ulid6 = monotonicUlid.generate();
59   * byte[] ulid7 = monotonicUlid.generateBinary();
60   * ULID ulid8 = monotonicUlid.generateULID();
61   *
62   * // Create a new independent instance of MonotonicULID using provided Random instance
63   * MonotonicULID monotonicUlid2 = new MonotonicULID(SecureRandom.newInstance("SHA1PRNG"));
64   * String ulid9 = monotonicUlid2.generate();
65   * // Subsequent calls generates monotonic ULID
66   * String ulid10 = monotonicUlid2.generate();
67   * byte[] ulid11 = monotonicUlid2.generateBinary();
68   * ULID ulid12 = monotonicUlid2.generateULID();
69   * </pre>
70   *
71   * @author azam
72   * @since 1.0.3
73   *
74   * @see <a href="https://github.com/azam">ulidj</a>
75   * @see <a href="https://github.com/ulid/spec">ULID Specification</a>
76   * @see <a href="https://www.crockford.com/wrmg/base32.html">Crockford Base32 Encoding</a>
77   */
78  public final class MonotonicULID {
79    private final Random random;
80    private final byte[] lastEntropy;
81    private long lastTimestamp;
82  
83    /**
84     * This allows lazy initialization of the default {@link java.util.Random} instance, backed by
85     * {@link java.security.SecureRandom} instance.
86     *
87     * @since 1.1.0
88     */
89    private static class LazyDefaults {
90      /**
91       * Default {@link java.util.Random} instance.
92       *
93       * @since 1.1.0
94       */
95      static final Random random = new SecureRandom();
96  
97      /**
98       * Default {@link io.azam.ulidj.MonotonicULID} instance.
99       *
100      * @since 1.1.0
101      */
102     static final MonotonicULID singleton = new MonotonicULID(random);
103   }
104 
105   /**
106    * Generate a monotonic ULID generator instance using default {@link java.util.Random} instance
107    * backed by {@link java.security.SecureRandom}.
108    *
109    * @since 1.0.3
110    */
111   public MonotonicULID() {
112     this(LazyDefaults.random);
113   }
114 
115   /**
116    * Generate a monotonic ULID generator instance.
117    *
118    * @param random {@link java.util.Random} instance
119    * @since 1.0.3
120    */
121   public MonotonicULID(Random random) {
122     this.random = random == null ? LazyDefaults.random : random;
123     this.lastEntropy = new byte[ULID.ENTROPY_LENGTH];
124     this.lastTimestamp = 0L;
125   }
126 
127   /**
128    * Generate ULID string monotonically. If this method is called within the same millisecond, last
129    * entropy will be incremented by 1 and the ULID string of incremented value is returned.<br>
130    * <br>
131    * This method will throw a {@link java.lang.IllegalStateException} exception if incremented value
132    * overflows entropy length (80-bits/10-bytes).
133    *
134    * @return ULID string
135    * @throws IllegalStateException if time is out of ULID specification or entropy overflowed
136    * @since 1.0.3
137    */
138   public synchronized String generate() {
139     mutate();
140     return ULID.generate(this.lastTimestamp, this.lastEntropy);
141   }
142 
143   /**
144    * Generate ULID binary monotonically. If this method is called within the same millisecond, last
145    * entropy will be incremented by 1 and the ULID string of incremented value is returned.<br>
146    * <br>
147    * This method will throw a {@link java.lang.IllegalStateException} exception if incremented value
148    * overflows entropy length (80-bits/10-bytes).
149    *
150    * @return ULID binary
151    * @throws IllegalStateException if time is out of ULID specification or entropy overflowed
152    * @since 1.0.4
153    */
154   public synchronized byte[] generateBinary() {
155     mutate();
156     return ULID.generateBinary(this.lastTimestamp, this.lastEntropy);
157   }
158 
159   /**
160    * Generate ULID instance monotonically. If this method is called within the same millisecond,
161    * last entropy will be incremented by 1 and the ULID string of incremented value is returned.<br>
162    * <br>
163    * This method will throw a {@link java.lang.IllegalStateException} exception if incremented value
164    * overflows entropy length (80-bits/10-bytes)
165    *
166    * @return ULID instance
167    * @throws IllegalStateException if time is out of ULID specification or entropy overflowed
168    * @since 1.1.0
169    */
170   public synchronized ULID generateULID() {
171     mutate();
172     return ULID.generateULID(this.lastTimestamp, this.lastEntropy);
173   }
174 
175   /**
176    * Check if the last timestamp is less than the current time. If so, generate new entropy
177    * otherwise increment the last entropy by 1-bit in the least significant bit with carryover. If
178    * the last entropy overflows, revert to the previous value and throw an exception.
179    *
180    * @throws IllegalStateException if time is out of ULID specification or entropy overflowed
181    * @since 1.1.0
182    */
183   private void mutate() {
184     long now = System.currentTimeMillis();
185     if (now < ULID.MIN_TIME || now > ULID.MAX_TIME)
186       throw new IllegalStateException("Time is out of ULID specification");
187     if (now <= this.lastTimestamp) {
188       // Keep last timestamp and increment last entropy
189       // Save a copy of last entropy in case of entropy overflow
190       byte[] previousEntropy = new byte[ULID.ENTROPY_LENGTH];
191       System.arraycopy(this.lastEntropy, 0, previousEntropy, 0, ULID.ENTROPY_LENGTH);
192       // Entropy is big-endian (network byte order) per ULID spec
193       // Increment last entropy by 1
194       boolean carry = true;
195       for (int i = ULID.ENTROPY_LENGTH - 1; i >= 0; i--) {
196         if (carry) {
197           byte work = this.lastEntropy[i];
198           work = (byte) (work + 0x01);
199           carry = this.lastEntropy[i] == (byte) 0xff;
200           this.lastEntropy[i] = work;
201         }
202       }
203       // Last byte has carry over
204       if (carry) {
205         // Revert last entropy to previous value
206         System.arraycopy(previousEntropy, 0, this.lastEntropy, 0, ULID.ENTROPY_LENGTH);
207         // Throw error if entropy overflows in same millisecond per ULID spec
208         throw new IllegalStateException("ULID entropy overflowed for same millisecond");
209       }
210     } else {
211       // Generate new entropy
212       this.lastTimestamp = now;
213       this.random.nextBytes(this.lastEntropy);
214     }
215   }

217   /**
218    * Generate ULID string monotonically with the defauit {@link io.azam.ulidj.MonotonicULID}
219    * instance backed by {@link java.security.SecureRandom}. If this method is called within the same
220    * millisecond, last entropy will be incremented by 1 and the ULID string of incremented value is
221    * returned.<br>
222    * <br>
223    * This method will throw a {@link java.lang.IllegalStateException} exception if incremented value
224    * overflows entropy length (80-bits/10-bytes).
225    *
226    * @return ULID string
227    * @throws IllegalStateException if time is out of ULID specification or entropy overflowed
228    * @since 1.0.3
229    */
230   public static String random() {
231     return LazyDefaults.singleton.generate();
232   }

237    * millisecond, last entropy will be incremented by 1 and the ULID string of incremented value is

239    * <br>
240    * This method will throw a {@link java.lang.IllegalStateException} exception if incremented value

243    * @return ULID binary
244    * @throws IllegalStateException if time is out of ULID specification or entropy overflowed
245    * @since 1.0.4
246    */
247   public static byte[] randomBinary() {
248     return LazyDefaults.singleton.generateBinary();
249   }

251   /**
252    * Generate ULID instance monotonically with the defauit {@link io.azam.ulidj.MonotonicULID} *
253    * instance backed by {@link java.security.SecureRandom}. If this method is called within the same
254    * millisecond, last entropy will be incremented by 1 and the ULID string of incremented value is

256    * <br>

261    * @throws IllegalStateException if time is out of ULID specification or entropy overflowed
262    * @since 1.1.0
263    */
264   public static ULID randomULID() {
265     return LazyDefaults.singleton.generateULID();
266   }
